<body>

    <div id="character"></div>

    <script>

        // Select the character element

        const character = document.getElementById('character');


        // Function to perform hyperjump

        function hyperjump() {

            character.classList.add('hyperjump');

            // Remove the class after the animation

            setTimeout(() => character.classList.remove('hyperjump'), 500);

        }


        // Add event listener for the jump

        character.addEventListener('click', hyperjump);

    </script>
